from PIL import Image

def encrypt_image(image_path, key):
    img = Image.open(image_path)
    pixels = img.load()

    width, height = img.size

    for x in range(width):
        for y in range(height):
            r, g, b = pixels[x, y]

            r = (r + key) % 256
            g = (g + key) % 256
            b = (b + key) % 256

            pixels[x, y] = (r, g, b)

    encrypted_path = "encrypted_image.png"
    img.save(encrypted_path)
    print(f"Encrypted image saved as {encrypted_path}")


def decrypt_image(image_path, key):
    img = Image.open(image_path)
    pixels = img.load()

    width, height = img.size

    for x in range(width):
        for y in range(height):
            r, g, b = pixels[x, y]

            r = (r - key) % 256
            g = (g - key) % 256
            b = (b - key) % 256

            pixels[x, y] = (r, g, b)

    decrypted_path = "decrypted_image.png"
    img.save(decrypted_path)
    print(f"Decrypted image saved as {decrypted_path}")


choice = input("Enter 'E' to Encrypt or 'D' to Decrypt: ").upper()
image_path = input("Enter image path: ")
key = int(input("Enter encryption key (0-255): "))

if choice == 'E':
    encrypt_image(image_path, key)
elif choice == 'D':
    decrypt_image(image_path, key)
else:
    print("Invalid choice.")
